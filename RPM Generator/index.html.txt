        function parseWaktu(waktuStr) {
            if (!waktuStr) return 70;
            const match = waktuStr.toLowerCase().match(/(\d+)\s*[x×]\s*(\d+)/);
            if (match) return parseInt(match[1]) * parseInt(match[2]);
            const numMatch = waktuStr.match(/(\d+)/);
            if (numMatch) return parseInt(numMatch[1]);
            return 70;
        }

        function round5(x) { return Math.round(x/5)*5; }

        function generateSteps(tp, modelStr, waktuStr, mitra, lingkungan, digital) {
            const totalMenit = parseWaktu(waktuStr);
            const waktuAwal = Math.max(5, round5(totalMenit * 0.15));
            const waktuAkhir = Math.max(5, round5(totalMenit * 0.15));
            let waktuInti = totalMenit - waktuAwal - waktuAkhir;

            let waktuMemahami = Math.max(5, round5(waktuInti * 0.35));
            let waktuMengaplikasi = Math.max(5, round5(waktuInti * 0.40));
            let waktuMerefleksi = waktuInti - waktuMemahami - waktuMengaplikasi;
            
            if(waktuMerefleksi < 5) {
                waktuMerefleksi = 5;
                waktuMengaplikasi = waktuInti - waktuMemahami - waktuMerefleksi;
            }

            const model = (modelStr || "").toLowerCase();
            const hasMitra = mitra && mitra !== '-';
            const hasLingkungan = lingkungan && lingkungan !== '-';
            const hasDigital = digital && digital !== '-';

            let awal = [
                "Guru membuka pelajaran dengan salam, doa, dan memeriksa kehadiran murid.",
                "Guru memberikan apersepsi untuk mengaitkan materi sebelumnya dengan tujuan pembelajaran hari ini.",
                "Guru menyampaikan tujuan pembelajaran dan tahapan kegiatan yang akan dilakukan."
            ];
            if (hasDigital) awal.push(`Guru melakukan ice breaking atau kuis singkat menggunakan ${digital} untuk membangun suasana menggembirakan.`);

            let memahami, mengaplikasi, merefleksi;

            if (model.includes("pbl") || model.includes("problem based")) {
                memahami = ["Guru menyajikan masalah kontekstual yang berkaitan dengan tujuan pembelajaran (Orientasi Masalah).", "Murid mengamati dan mengidentifikasi masalah tersebut melalui diskusi kelompok untuk membangun pemahaman awal."];
                mengaplikasi = ["Murid berkolaborasi dalam kelompok untuk merumuskan hipotesis dan mencari solusi pemecahan masalah secara mandiri (Mengorganisasikan Belajar).", "Guru berperan sebagai fasilitator yang mendampingi proses diskusi dan pencarian solusi."];
                merefleksi = ["Perwakilan kelompok mempresentasikan hasil solusi, kelompok lain memberikan tanggapan.", "Guru dan murid menyepakati jawaban terbaik dan merefleksikan makna dari proses pemecahan masalah tersebut (Menganalisis & Mengevaluasi)."];
            } else if (model.includes("discovery") || model.includes("penemuan")) {
                memahami = ["Guru memberikan stimulus berupa data atau fenomena terkait tujuan pembelajaran.", "Murid mengidentifikasi dan merumuskan pertanyaan/persoalan berdasarkan data yang diamati (Statement & Question)."];
                mengaplikasi = ["Murid melakukan eksplorasi dan percobaan (proses penemuan) untuk mengumpulkan informasi dan membuktikan hipotesis (Experimentation).", "Murid mencatat data hasil percobaan secara mandiri atau kelompok."];
                merefleksi = ["Murid menyimpulkan hasil penemuan mereka berdasarkan data yang dikumpulkan.", "Guru mengkonfirmasi kesimpulan murid dan memberikan penguatan konsep yang ditemukan (Conclusion Making)."];
            } else if (model.includes("kooperatif") || model.includes("diskusi") || model.includes("kelompok")) {
                memahami = ["Guru membagi murid ke dalam kelompok-kelompok kooperatif dan membagikan bahan ajar.", "Murid membaca dan mendiskusikan bahan ajar secara berkelompok untuk saling memahami konsep dasar (Group Discussion)."];
                mengaplikasi = ["Setiap anggota kelompok bertanggung jawab atas bagian tertentu, kemudian saling mengajarkan kepada anggota kelompok lainnya (Peer Teaching).", "Guru memantau aktifitas kelompok dan memastikan setiap murid berkontribusi."];
                merefleksi = ["Guru mengevaluasi pemahaman individu dan kelompok terkait materi yang dipelajari.", "Murid merefleksikan manfaat bekerja sama dan pembelajaran yang telah dicapai (Group Evaluation)."];
            } else {
                memahami = ["Guru mengajak murid melakukan observasi atau diskusi awal terkait tujuan pembelajaran untuk membangun kesadaran konsep.", "Murid mengeksplorasi materi dengan suasana yang menggembirakan melalui media interaktif."];
                mengaplikasi = ["Murid menerapkan pengetahuan dalam situasi nyata atau simulasi kelompok secara bermakna.", "Murid memastikan setiap proses pengerjaan tugas dilakukan dengan penuh kesadaran (mindfulness)."];
                merefleksi = ["Guru dan murid melakukan refleksi bersama untuk menemukan makna dari pembelajaran yang baru dilalui.", "Murid mengungkapkan perasaan dan insight yang didapat dalam jurnal singkat atau diskusi kelas."];
            }

            if (hasLingkungan) memahami.push(`Murid melakukan kegiatan eksplorasi/observasi dengan memanfaatkan lingkungan ${lingkungan}.`);
            if (hasDigital) memahami.push(`Murid menggunakan ${digital} untuk membantu mencari informasi dan memahami konsep.`);
            if (hasMitra) mengaplikasi.push(`Murid berdiskusi atau berkolaborasi dengan ${mitra} untuk memperdalam penerapan materi.`);
            if (hasDigital) mengaplikasi.push(`Murid mendokumentasikan atau mengerjakan tugas penerapan menggunakan ${digital}.`);
            if (hasMitra) merefleksi.push(`Murid mendapatkan umpan balik atau berbagi pengalaman dengan ${mitra} terkait proses pembelajaran.`);
            if (hasDigital) merefleksi.push(`Guru dan murid menggunakan ${digital} untuk melakukan kuis refleksi atau evaluasi singkat.`);

            let akhir = [
                "Guru bersama murid menyimpulkan materi pembelajaran yang telah dipelajari.",
                "Guru melakukan asesmen formatif untuk mengecek pemahaman murid.",
                "Guru memberikan tindak lanjut berupa penguatan atau tugas pengayaan.",
                "Guru menutup pelajaran dengan salam."
            ];

            return [
                { tahap: "Kegiatan Awal", aktivitas: awal, waktu: `${waktuAwal} menit` },
                { tahap: "Kegiatan Inti - Memahami", aktivitas: memahami, waktu: `${waktuMemahami} menit` },
                { tahap: "Kegiatan Inti - Mengaplikasi", aktivitas: mengaplikasi, waktu: `${waktuMengaplikasi} menit` },
                { tahap: "Kegiatan Inti - Merefleksi", aktivitas: merefleksi, waktu: `${waktuMerefleksi} menit` },
                { tahap: "Kegiatan Akhir", aktivitas: akhir, waktu: `${waktuAkhir} menit` }
            ];
        }

            const date = new Date(dateStr + 'T00:00:00');
            return date.toLocaleDateString('id-ID', { day: 'numeric', month: 'long', year: 'numeric' });
        }

        function copyRPM() {
            const btn = document.getElementById('btnCopy');
            const contentEl = document.getElementById('previewContent');
            const container = document.createElement('div');
            container.innerHTML = contentEl.innerHTML;
            container.style.fontFamily = "'Times New Roman', Times, serif";
            container.style.lineHeight = "1.6";
            container.style.color = "#000";
            container.style.position = 'fixed';
            container.style.left = '-9999px';
            container.style.top = '0';
            document.body.appendChild(container);

            const range = document.createRange();
            range.selectNodeContents(container);
            const selection = window.getSelection();
            selection.removeAllRanges();
            selection.addRange(range);

            try {
                document.execCommand('copy');
                showSuccessMessage("Dokumen RPM berhasil disalin! Paste ke Word/Docs.");
                btn.innerHTML = '<i data-lucide="check" class="w-5 h-5"></i><span>Tersalin!</span>';
                lucide.createIcons();
                setTimeout(() => {
                    btn.innerHTML = '<i data-lucide="clipboard-copy" class="w-5 h-5"></i><span>Salin Format RPM</span>';
                    lucide.createIcons();
                }, 2500);
            } catch (err) {
                alert("Gagal menyalin teks. Silakan blok teks secara manual lalu Ctrl+C.");
            } finally {
                selection.removeAllRanges();
                document.body.removeChild(container);
            }
        }

        async function downloadAsWord() {
            const btn = document.getElementById('btnWord');
            btn.disabled = true;
            btn.innerHTML = '<i class="animate-spin" data-lucide="loader-2"></i> Memproses...';
            lucide.createIcons();

            const contentEl = document.getElementById('previewContent');
            const fileName = `RPM_${currentData.mapel}_${currentData.materi}.doc`;
            
            const htmlContent = `
                <html xmlns:o='urn:schemas-microsoft-com:office:office' xmlns:w='urn:schemas-microsoft-com:office:word' xmlns='http://www.w3.org/TR/REC-html40'>
                <head>
                    <meta charset='utf-8'>
                    <style>
                        body { font-family: 'Times New Roman', serif; font-size: 12pt; line-height: 1.6; color: #000; }
                        table { border-collapse: collapse; width: 100%; }
                        th, td { border: 1px solid black; padding: 6px; vertical-align: top; }
                        h1, h2 { font-family: 'Times New Roman', serif; }
                    </style>
                </head>
                <body>
                    ${contentEl.innerHTML}
                </body>
                </html>
            `;

            const blob = new Blob(['\ufeff', htmlContent], { type: 'application/msword' });

            // Cek apakah browser mendukung File System Access API (Save As Dialog)
            if (window.showSaveFilePicker) {
                try {
                    const handle = await window.showSaveFilePicker({
                        suggestedName: fileName,
                        types: [{
                            description: 'Word Document',
                            accept: { 'application/msword': ['.doc'] },
                        }],
                    });
                    const writable = await handle.createWritable();
                    await writable.write(blob);
                    await writable.close();
                    showSuccessMessage("Dokumen Word berhasil disimpan!");
                } catch (err) {
                    // Error AbortError terjadi jika user menekan Cancel/Batal pada jendela Save As
                    if (err.name !== 'AbortError') {
                        console.error(err);
                        // Fallback jika ada error lain
                        saveAs(blob, fileName);
                        showSuccessMessage("Dokumen disimpan di folder Downloads.");
                    }
                }
            } else {
                // Fallback untuk browser yang tidak support Save As (seperti Firefox / file offline)
                saveAs(blob, fileName);
                showSuccessMessage("Dokumen disimpan di Downloads. (Aktifkan 'Tanyakan lokasi penyimpanan' di pengaturan browser untuk memilih folder).");
            }

            btn.disabled = false;
            btn.innerHTML = '<i data-lucide="file-text" class="w-5 h-5"></i> Simpan Sebagai Word';
            lucide.createIcons();
        }
